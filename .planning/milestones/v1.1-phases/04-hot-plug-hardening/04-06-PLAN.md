```
#if defined(_WIN32)
#include <catch2/catch_test_macros.hpp>
#include "ajazz/core/hotplug_monitor.hpp"
// ... include the AJAZZ_TESTING-gated parseDevicePathW header ...
using namespace ajazz::core;

TEST_CASE("Win32 device-path parses VID/PID/serial correctly", "[hotplug][win32]") {
    auto ev = parseDevicePathW(L"\\\\?\\HID#VID_5548&PID_6672#7&deadbeef&0&0000", HotplugAction::Arrived);
    REQUIRE(ev.vid == 0x5548);
    REQUIRE(ev.pid == 0x6672);
    REQUIRE(ev.action == HotplugAction::Arrived);
    REQUIRE(ev.serial.find("deadbeef") != std::string::npos);
}

TEST_CASE("Win32 injectEvent path round-trips a synthetic event end-to-end", "[hotplug][win32]") {
    HotplugEvent observed{};
    bool fired = false;
    HotplugMonitor mon{[&](HotplugEvent const& ev){ observed = ev; fired = true; }};
    // Do NOT call start() — we only want the in-memory dispatch path.
    mon.injectEvent(HotplugEvent{HotplugAction::Removed, 0x5548, 0x6672, "deadbeef"});
    REQUIRE(fired);
    REQUIRE(observed.vid == 0x5548);
    REQUIRE(observed.pid == 0x6672);
    REQUIRE(observed.action == HotplugAction::Removed);
}
#endif
```
